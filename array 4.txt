#include <stdio.h>
#include <stdlib.h>
#include <time.h>

#define TAM 50

void cargarVector(int v[]);
int contarMayor(int v[]);

int main() {

    int vector[TAM];
    int repeticiones;

    cargarVector(vector);

    repeticiones = contarMayor(vector);

    printf("El mayor valor se repite %d veces.\n", repeticiones);

    return 0;
}

void cargarVector(int v[]) {

    srand(time(NULL));

    for(int i = 0; i < TAM; i++){
        v[i] = rand() % 11 + 10;   // genera valores entre 10 y 20
    }
}

int contarMayor(int v[]) {

    int mayor = v[0];
    int contador = 0;

    // Buscar el mayor
    for(int i = 1; i < TAM; i++){
        if(v[i] > mayor){
            mayor = v[i];
        }
    }

    // Contar cu√°ntas veces aparece
    for(int i = 0; i < TAM; i++){
        if(v[i] == mayor){
            contador++;
        }
    }

    return contador;
}
